ذُوقُوا فِتْنَتَكُمْ

الْقَوْل فِي تَأْوِيل قَوْله تَعَالَى : { ذُوقُوا فِتْنَتَكُمْ } يَعْنِي تَعَالَى ذِكْرُهُ بِقَوْلِهِ : { ذُوقُوا فِتْنَتَكُمْ } يُقَال لَهُمْ : ذُوقُوا فِتْنَتَكُمْ وَتَرَكَ يُقَال لَهُمْ لِدَلَالَةِ الْكَلَام عَلَيْهَا. وَيَعْنِي بِقَوْلِهِ : { فِتْنَتَكُمْ } : عَذَابَكُمْ وَحَرِيقَكُمْ . وَاخْتَلَفَ أَهْل التَّأْوِيل فِي ذَلِكَ , فَقَالَ بُغْضهمْ بِالَّذِي قُلْنَا فِيهِ . ذِكْرُ مَنْ قَالَ ذَلِكَ : 24848 - حَدَّثَنِي مُحَمَّد بْن عَمْرو , قَالَ : ثنا أَبُو عَاصِم , قَالَ : ثنا عِيسَى ; وَحَدَّثَنِي الْحَارِث , قَالَ : ثنا الْحَسَن , قَالَ : ثنا وَرْقَاء جَمِيعًا , عَنِ ابْن أَبِي نَجِيح , عَنْ مُجَاهِد , قَوْله : { فِتْنَتكُمْ } قَالَ : حَرِيقكُمْ . 24849 - حَدَّثَنَا بِشْر , قَالَ : ثنا يَزِيد , قَالَ : ثنا سَعِيد , عَنْ قَتَادَة { ذُوقُوا فِتْنَتكُمْ } : ذُوقُوا عَذَابكُمْ } هَذَا الَّذِي كُنْتُمْ بِهِ تَسْتَعْجِلُونَ } . * - حَدَّثَنَا ابْن عَبْد الْأَعْلَى , قَالَ : ثنا ابْن ثَوْر , عَنْ مَعْمَر , عَنْ قَتَادَة , فِي قَوْله : { ذُوقُوا فِتْنَتكُمْ } يَقُول : يَوْم يُعَذَّبُونَ , فَيَقُول : ذُوقُوا عَذَابكُمْ . 24850 - حُدِّثْت عَنِ الْحُسَيْن , قَالَ : سَمِعْت أَبَا مُعَاذ يَقُول : أَخْبَرَنَا عُبَيْد , قَالَ : سَمِعْت الضَّحَّاك يَقُول فِي قَوْله : { ذُوقُوا فِتْنَتَكُمْ } يَقُول : حَرِيقَكُمْ . 24851 - حَدَّثَنِي ابْن حُمَيْد , قَالَ : ثنا مِهْرَان , عَنْ سُفْيَان { ذُوقُوا فِتْنَتَكُمْ } يَقُول : احْتِرَاقَكُمْ . 24852 - حَدَّثَنِي يُونُس , قَالَ : أَخْبَرَنَا ابْن وَهْب , قَالَ : قَالَ ابْن زَيْد , فِي قَوْله : { ذُوقُوا فِتْنَتَكُمْ } قَالَ : ذُوقُوا عَذَابَكُمْ . وَقَالَ آخَرُونَ : عَنَى بِذَلِكَ : ذُوقُوا تَعْذِيبَكُمْ أَوْ كَذِبَكُمْ . ذِكْرُ مَنْ قَالَ ذَلِكَ : 24853 - حَدَّثَنِي مُحَمَّد بْن سَعْد , قَالَ : ثني أَبِي , قَالَ : ثني عَمِّي , قَالَ : ثني أَبِي , عَنْ أَبِيهِ , عَنِ ابْن عَبَّاس , قَوْله : { ذُوقُوا فِتْنَتَكُمْ } يَقُول : تَكْذِيبَكُمْ . 24854 - حُدِّثْت عَنِ الْحُسَيْن , قَالَ : سَمِعْت أَبَا مُعَاذ يَقُول : أَخْبَرَنَا عُبَيْد , قَالَ : سَمِعْت الضَّحَّاك يَقُول فِي قَوْله : { ذُوقُوا فِتْنَتَكُمْ } يَقُول : حَرِيقَكُمْ , وَيُقَال : كَذِبَكُمْ .

هَذَا الَّذِي كُنْتُمْ بِهِ تَسْتَعْجِلُونَ

وَقَوْله : { هَذَا الَّذِي كُنْتُمْ بِهِ تَسْتَعْجِلُونَ } يَقُول تَعَالَى ذِكْرُهُ : يُقَال لَهُمْ : هَذَا الْعَذَاب الَّذِي تُوَفَّوْنَهُ الْيَوْم , هُوَ الْعَذَاب الَّذِي كُنْتُمْ بِهِ تَسْتَعْجِلُونَ فِي الدُّنْيَا .
